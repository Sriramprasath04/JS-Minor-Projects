9) Write a switch statement to print the month in a quater
                Quater1: Jan, Feb, Mar
                Quater2: Apr, May, Jun
                Quater3: Jul, Aug, Sep
                Quater4: Oct, Nov, Dec
Answer:
  let quat = 1;
  switch(quat){
      case 1:
          console.log("Jan, Feb, Mar");
          break;
      case 2:
          console.log("Apr, May, Jun");
          break;
      case 3:
          console.log("Jul, Aug, Sep");
          break;
      case 4:
          console.log("Oct, Nov, Dec");
          break;
      default:
          console.log("Not a Quater!");
  
  }
